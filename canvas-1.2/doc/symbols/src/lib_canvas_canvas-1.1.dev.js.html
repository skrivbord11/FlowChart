 23 /**
 24  * The canvas library class.
 25  *
 26  * HTML canvas library is a full-featured lightweight wrapper library of the
 27  * native html canvas element written in Javascript, aimed to make visualization
 28  * and animation using canvas simpler. Features animation support, layers, event
 29  * capture, multitouch and many examples.
 30  *
 31  * This is the main constructor.
 32  *
 33  * The parameters layerParent and drawOnly are used by layers, you dont have to
 34  * define these yourself.
 35  *
 36  * The fps parameter can have three kind of values
 37  * > 0: static image, the render function is called only once
 38  * > -1: animation without framerate limit, runs as fast as possible
 39  * > positive: for example 60, use to try to maintain given framerate
 40  *
 41  * @constructor
 42  * @param {string|object} canvas Canvas element id or the actual dom-element
 43  * @param {integer} [fps=0] Requested frames-per-second, use for animations, optional, defaults to 0
 44  * @param {function} [renderCallback=null] The function to call to render each frame, gets the canvas as "this", optional, can set later with setRenderCallback()
 45  * @param {boolean} [autostart=true] Should the rendering be started at once or manually later by calling start(), optional, defaults to true
 46  * @param {Canvas} [layerParent=null] The parent canvas element of given layer, used by layers system, no need to define
 47  * @param {boolean} [drawOnly=false] Is this a render-to-canvas operation canvas, does not intercept any events, no need to define
 48  * @version 1.1
 49  */

393 	 * The keyboard events storage object, use to get keyboard state.
394 	 *
395 	 * This stores the keys that are currently pressed, including the modifier
396 	 * keys ctrl, shift and alt. Use the kb.isDown(keyCode) method to check

401 	 * Has keys:
402 	 * > down: array of keycodes of buttons that are pressed

404 	 * > shift: is the SHIFT button pressed

416         shift : false,

419             for (var i = 0; i < this.down.length; i++) {
420                 if (this.down[i]['code'] == keyCode) {
421                     return true;
422                 }
423             }
424 
425             return false;
426         }

454 	/**
455 	 * Initializes the instance, this is called automatically.
456 	 *
457 	 * This method:
458 	 * > initializes the canvas context
459 	 * > registers and handles all events

461 	 * > starts the animation is autostart is true
462 	 *
463 	 * @type string
464 	 * @private
465 	 * @since 1.0
466 	 */
467     this.init = function() {
468 		if (renderCallback != null) {
469 			this.setRenderCallback(renderCallback);
470 		}
471 
472 		if (typeof(canvas) == 'object') {
473 			this.canvasElement = canvas;
474 		} else if (typeof(canvas) == 'string') {
475 			this.canvasElement = document.getElementById(canvas);
476 		} else {
477 			this.handleError('Invalid canvas reference given, expected canvas element or id');
478 
479 			return;
480 		}
481 
482         if (this.canvasElement == null) {
483             this.handleError('Canvas element could not be found');
484 
485             return;
486         }
487 
488         this.c = this.canvasElement.getContext(this.canvasContext);
489 
490         if (this.c == null) {
491             this.handleError('Given canvas element does not seem to be of canvas type, context not found');
492 
493             return;
494         }
495 
496 		if (typeof(canvas) == 'object') {
497 			this.width = canvas.width;
498 			this.height = canvas.height;
499 		} else {
500 			this.width = this.getWidth();
501 			this.height = this.getHeight();
502 
503 			this.canvasElement.width = this.width;
504 			this.canvasElement.height = this.height;
505 		}
506 
507         var self = this;
508 
509 		if (this.drawOnly != true) {
510 			// handle window resize only for root layer
511 			if (this.layerParent == null) {
512 				window.onresize = function() {
513 					self.handleWindowResize();
514 				}
515 
516 				document.onkeydown = function(event) {
517 					if (event == null) {
518 						event = window.event;
519 					}
520 
521 					var keyCode = event.keyCode;
522 					var character = event.which == null ? String.fromCharCode(event.keyCode) : String.fromCharCode(event.which);
523 					var isCtrlPressed = event.ctrlKey ? true : false;
524 					var isShiftPressed = event.shiftKey ? true : false;
525 					var isAltPressed = event.altKey ? true : false;
526 
527 					if (!self.kb.isDown(keyCode)) {
528 						self.kb['down'].push({'code' : keyCode, 'character' : character});
529 					}
530 
531 					self.kb['ctrl'] = isCtrlPressed;
532 					self.kb['shift'] = isShiftPressed;
533 					self.kb['alt'] = isAltPressed;
534 
535 					for (var layerName in self.layers) {
536 						self.layers[layerName].kb = self.kb;
537 						self.layers[layerName].onKeyDown(keyCode, character, isCtrlPressed, isShiftPressed, isAltPressed);
538 					}
539 
540 					return self.onKeyDown(keyCode, character, isCtrlPressed, isShiftPressed, isAltPressed);
541 				};
542 
543 				document.onkeyup = function(event) {
544 					if (event == null) {
545 						event = window.event;
546 					}
547 
548 					var keyCode = event.keyCode;
549 					var character = event.which == null ? String.fromCharCode(event.keyCode) : String.fromCharCode(event.which);
550 					var isCtrlPressed = event.ctrlKey ? true : false;
551 					var isShiftPressed = event.shiftKey ? true : false;
552 					var isAltPressed = event.altKey ? true : false;
553 
554 					var newKbDown = [];
555 
556 					for (var i = 0; i < self.kb['down'].length; i++) {
557 						if (self.kb['down'][i]['code'] != keyCode) {
558 							newKbDown.push({'code' : self.kb['down'][i]['code'], 'character' : self.kb['down'][i]['character']});
559 						}
560 					}
561 
562 					self.kb['down'] = newKbDown;
563 					self.kb['ctrl'] = isCtrlPressed;
564 					self.kb['shift'] = isShiftPressed;
565 					self.kb['alt'] = isAltPressed;
566 
567 					for (var layerName in self.layers) {
568 						self.layers[layerName].kb = self.kb;
569 						self.layers[layerName].onKeyUp(keyCode, character, isCtrlPressed, isShiftPressed, isAltPressed);
570 					}
571 
572 					return self.onKeyUp(keyCode, character, isCtrlPressed, isShiftPressed, isAltPressed);
573 				}
574

577 						event = window.event;
578 					}
579

592 				}

595 			this.canvasElement.addEventListener('touchstart', function(event) {
596 				var touchCount = event.changedTouches.length;
597 				var touches = [];
598 				var touch = null;
599 
600 				for (var i = 0; i < touchCount; i++) {
601 					touch = event.changedTouches[i];
602 
603 					var touchInfo = {
604 						pageX : touch.pageX,
605 						pageY : touch.pageY,
606 						clientX : touch.clientX,
607 						clientY : touch.clientY,
608 						screenX : touch.screenX,
609 						screenY : touch.screenY,
610 						target : touch.target,
611 						identifier : touch.identifier
612 					}
613 
614 					self.onTouchStart(touchInfo, i, touchCount, event);
615 
616 					touches.push(touchInfo);
617 
618 					self.previousTouchInfo[touch.identifier] = touchInfo;
619 				}
620 
621 				if (touchCount == 1) {
622 					touch = event.changedTouches[0];
623 
624 					var x = touch.clientX;
625 					var y = touch.clientY;
626 
627 					if (self.touchEmulatesMouse) {
628 						self.mouse.x = x;
629 						self.mouse.y = y;
630 						self.mouse.left = true;
631 
632 						if (self.layerParent != null) {
633 							self.layerParent.onMouseDown(x, y, 0);
634 						}
635 
636 						self.onMouseDown(x, y, 0);
637 					}
638 				} else {
639 					self.onMultiTouchStart(touches, event);
640 				}

741 			this.canvasElement.addEventListener('mousedown', function(event) {
742 				var x = null;
743 				var y = null;
744 
745 				if (typeof (event.layerX) != 'undefined') {
746 					x = event.layerX;
747 					y = event.layerY;
748 				} else if (typeof (event.offsetX) != 'undefined') {
749 					x = event.offsetX;
750 					y = event.offsetY;
751 				}
752 
753 				var button = typeof(event.which) != 'undefined' ? event.which : event.button;
754 
755 				if (button == 1) {
756 					self.mouse['left'] = true;
757 				} else if (button == 3) {
758 					self.mouse['right'] = true;
759 				} else if (button == 2) {
760 					self.mouse['middle'] = true;
761 				}
762
